#### 특정 범위에 속하는 요소찾기
```javascript
function filterRange(arr, a, b) {
return arr.filter(item => (a <= item && item <= b));
}

let arr = [5, 3, 8, 1];

let filtered = filterRange(arr, 1, 4);
alert( filtered ); 
alert( arr );
```

#### 특정 범위에 속하는 요소찾기
```javascript
function filterRangeInPlace(arr, a, b) {
  for (let i = 0; i < arr.length; i++) {
    let val = arr[i];
    if (val < a || val > b) {
      arr.splice(i, 1);
      i--;
    }
  }

}

let arr = [5, 3, 8, 1];
filterRangeInPlace(arr, 1, 4); 
alert( arr );
```
