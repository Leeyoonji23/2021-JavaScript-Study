### Q4. 최대합 부분 배열

```javascript
function getMaxSubSum(arr) {
    let maxSum = 0;
    let temp = 0;
    arr.forEach(v => {
        temp += v;
        if (temp < 0) temp = 0;
        else if(maxSum < temp) maxSum = temp;
    })
    return maxSum;
}
```

```javascript
function filterRangeInPlace(arr, start, end) {
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] < start || arr[i] > end) {
            arr.splice(i, 1);
            i--;
        }
    }
}
```
